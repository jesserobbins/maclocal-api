import Foundation

struct WeatherService {
    let apiKey: String
    let baseURL = "https://api.weather.example.com/v2"

    func fetchForecast(latitude: Double, longitude: Double) async throws -> Forecast {
        let url = URL(string: "\(baseURL)/forecast?lat=\(latitude)&lon=\(longitude)&key=\(apiKey)")!
        let (data, response) = try await URLSession.shared.data(from: url)
        guard let httpResponse = response as? HTTPURLResponse,
              httpResponse.statusCode == 200 else {
            throw WeatherError.networkFailure
        }
        return try JSONDecoder().decode(Forecast.self, from: data)
    }
}
